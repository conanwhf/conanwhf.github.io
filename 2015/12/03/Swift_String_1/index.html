下标（Index）操作

str.startIndex  //起始下标
str.endIndex        //结束下标
var i      = str.startIndex.successor() //下标的前后移动
str.endIndex.predecessor()
str[i]  //通过Index能直接拿到字符
str.characters[i]   //第二种方法
i          = i.advancedBy(2)        //起始位之后3个字符，即第四个字符
i.distanceTo(str.endIndex)
